Trait std::storage::storable_slice::StorableSlice

pub trait StorableSlice<T> {
    #[storage(read, write)]
    fn write_slice(self, argument: T);
    #[storage(read)]
    fn read_slice(self) -> Option<T>;
    #[storage(read, write)]
    fn clear(self) -> bool;
    #[storage(read)]
    fn len(self) -> u64;
}
