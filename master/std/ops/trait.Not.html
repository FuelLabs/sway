Trait to invert a type.

Required Methods

fn not(self) -> Self

Inverts the value of the type.

Returns

  • [Self] - The result of the inverse.

Examples

struct MyStruct {
    val: bool,
}

impl Not for MyStruct {
    fn not(self) -> Self {
        Self {
            val: !self.val,
        }
    }
}

fn foo() {
    let struct = MyStruct { val: true };
    let result_struct = !struct;
    assert(!result_struct.val);
}
